<script>
const ONDERDEEL_MAP = {
  'oefenen':  'Oefenen!',
  'zoeje':    'Zo doe je dat!',
  'hoever':   'Hoe ver sta ik?',
  'hoeveR':   'Hoe ver sta ik?',
  'ontdek':   'Ontdekkingstocht',
};

function handleDrop(e) {
  e.preventDefault();
  document.getElementById('upload-zone').classList.remove('over');
  const bestand = e.dataTransfer.files[0];
  if (bestand) handleBestand(bestand);
}

function handleBestand(bestand) {
  if (!bestand || !bestand.name.endsWith('.html')) {
    toonMelding('Kies een .html bestand.', 'fout');
    return;
  }
  const reader = new FileReader();
  reader.onload = e => verwerkInhoud(e.target.result);
  reader.readAsText(bestand);
}

function verwerkInhoud(tekst) {
  window._tekst = tekst;
  const beginPagina = parseInt(document.getElementById('inp-beginpagina').value) || 453;

  // Verzamel posities van alle paginawissels
  const startMarker = tekst.indexOf('/* BEGIN VANAF HIER MET TELLEN */');
  const startPositie = startMarker !== -1 ? startMarker : 0;

  const paginaRegex = /nieuwePagina\s*\(\s*\)|forceerPaginawissel\s*\(\s*\)/g;
  const paginaPosities = [];
  let pm;
  while ((pm = paginaRegex.exec(tekst)) !== null) {
    if (pm.index >= startPositie) paginaPosities.push(pm.index);
  }

  // Zoek alle laadOefeningen aanroepen
  const regex = /laadOefeningen\(\s*'([^']+)'\s*,\s*\[([\s\S]*?)\]\s*\)/g;
  let m;
  const oefeningen = [];
  let teller = 0;

  while ((m = regex.exec(tekst)) !== null) {
    const positie   = m.index;
    const containerId   = m[1];
    const bestandenBlok = m[2];

    // Tel paginawissels vóór deze positie
    const aantalPaginawissels = paginaPosities.filter(p => p < positie).length;
    const bladzijde = beginPagina + aantalPaginawissels - 1;

    const bestandRegex = /'([^']+\.js)'/g;
    let b;
    while ((b = bestandRegex.exec(bestandenBlok)) !== null) {
      teller++;
      const pad = b[1];
      const bestandsnaam = pad.split('/').pop().replace('.js', '');

      // Paragraaf uit container-id
      const paragraafMatch = containerId.match(/content-(\d)(\d+)-/);
      let paragraaf = '';
      if (paragraafMatch) {
        paragraaf = paragraafMatch[1] + '.' + parseInt(paragraafMatch[2]);
      }

      // Onderdeel uit container-id
      const onderdeelMatch = containerId.match(/content-\d+-([\w]+?)(-\d+)?$/);
      let onderdeel = '';
      const sleutelRaw = onderdeelMatch ? onderdeelMatch[1] : '';
      if (onderdeelMatch) {
        onderdeel = ONDERDEEL_MAP[sleutelRaw] || sleutelRaw;
      }

      // Filter op relevante onderdelen
      const toegestaan = ['oefenen', 'zoeje', 'hoeveR', 'hoever', 'ontdek'];
      if (!toegestaan.includes(sleutelRaw)) continue;

      oefeningen.push({ nr: teller, bestandsnaam, pad, onderdeel, paragraaf, bladzijde, containerId });
    }
  }

  if (oefeningen.length === 0) {
    toonMelding('Geen laadOefeningen() aanroepen gevonden.', 'fout');
    return;
  }

  oefeningen.forEach((o, i) => o.nr = i + 1);
  toonResultaat(oefeningen);
  toonMelding(oefeningen.length + ' oefeningen gevonden.', 'ok');
}

function toonResultaat(oefeningen) {
  const tbody = document.getElementById('tabel-body');
  tbody.innerHTML = '';

  // Groepeer per paragraaf
  const perParagraaf = {};
  oefeningen.forEach(o => {
    if (!perParagraaf[o.paragraaf]) perParagraaf[o.paragraaf] = [];
    perParagraaf[o.paragraaf].push(o);
  });

  Object.keys(perParagraaf).sort().forEach(par => {
    const trHeader = document.createElement('tr');
    trHeader.classList.add('paragraaf-header');
    trHeader.innerHTML = '<td colspan="6">Paragraaf ' + par + '</td>';
    tbody.appendChild(trHeader);

    perParagraaf[par].forEach(o => {
      const tr = document.createElement('tr');
      if (!o.bestandsnaam.startsWith('oe_')) tr.classList.add('in-bewerking');
      tr.innerHTML =
        '<td class="nr">' + o.nr + '</td>' +
        '<td class="bestand">' + o.bestandsnaam + '</td>' +
        '<td class="onderdeel">' + o.onderdeel + '</td>' +
        '<td class="container">' + o.paragraaf + '</td>' +
        '<td class="blz">' + o.bladzijde + '</td>' +
        '<td class="container">' + o.containerId + '</td>';
      tbody.appendChild(tr);
    });
  });

  // Statistieken
  const stats = document.getElementById('stats');
  const aantalPerOnderdeel = {};
  oefeningen.forEach(o => {
    aantalPerOnderdeel[o.onderdeel] = (aantalPerOnderdeel[o.onderdeel] || 0) + 1;
  });
  let statHTML = '<div class="stat-rij"><div class="stat"><strong>' + oefeningen.length + '</strong>Totaal</div>';
  Object.entries(aantalPerOnderdeel).forEach(function(e) {
    statHTML += '<div class="stat"><strong>' + e[1] + '</strong>' + e[0] + '</div>';
  });
  statHTML += '</div>';
  stats.innerHTML = statHTML;

  window._oefeningen = oefeningen;
  document.getElementById('resultaat-kaart').classList.add('zichtbaar');
  document.getElementById('resultaat-kaart').scrollIntoView({ behavior: 'smooth' });
}

function herbereken() {
  if (window._tekst) verwerkInhoud(window._tekst);
}

function downloadCSV() {
  if (!window._oefeningen) return;
  const rijen = [['Nr', 'Bestandsnaam', 'Pad', 'Onderdeel', 'Paragraaf', 'Bladzijde', 'Container']];
  window._oefeningen.forEach(o => {
    rijen.push([o.nr, o.bestandsnaam, o.pad, o.onderdeel, o.paragraaf, o.bladzijde, o.containerId]);
  });
  const csv = rijen.map(r => r.map(cel => '"' + cel + '"').join(';')).join('\n');
  const blob = new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8;' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'oefeningen_overzicht.csv';
  a.click();
  URL.revokeObjectURL(url);
}

function toonMelding(tekst, type) {
  const el = document.getElementById('melding');
  el.textContent = tekst;
  el.className = 'melding' + (type ? ' ' + type : '');
}
