let huidigePaginaNummer = EERSTE_PAGINA;

let eersteHoofdstukPagina = true;
const wrapper = document.getElementById('cursus-wrapper');

/** Maak een nieuwe pagina aan en voeg die toe aan de wrapper */
function nieuwePagina() {
  // Scheiding invoegen (systeem 2 toont dit, systeem 1 verbergt het via CSS)
  if (huidigePaginaEl !== null) {
    const scheiding = document.createElement('div');
    scheiding.classList.add('pagina-scheiding');
    scheiding.textContent = 'volgende pagina';
    wrapper.appendChild(scheiding);
  }

  const pagina = document.createElement('div');
  pagina.classList.add('pagina-blok');

  // Paginanummer (systeem 2)
  const nr = document.createElement('div');
  nr.classList.add('paginanummer');
  nr.textContent = huidigePaginaNummer;
  if (huidigePaginaNummer % 2 === 0) {
    nr.classList.add('links');
  } else {
    nr.classList.add('rechts');
  }
  pagina.appendChild(nr);

  // Hoofdstuktitel in header (alle pagina's behalve de allereerste)
  if (!eersteHoofdstukPagina) {
    const hdr = document.createElement('div');
    hdr.classList.add('hoofdstuk-titel-header');
    hdr.textContent = HOOFDSTUK_TITEL;
    // Even pagina: titel links; oneven pagina: titel rechts
    if (huidigePaginaNummer % 2 === 0) {
      hdr.classList.add('links');
    } else {
      hdr.classList.add('rechts');
    }
    pagina.appendChild(hdr);
  }

  wrapper.appendChild(pagina);
  huidigePaginaEl     = pagina;
  huidigePaginaNummer++;
  eersteHoofdstukPagina = false;
  return pagina;
}

/** Geeft het huidige inhoudselement terug (de actieve pagina) */
function huidigePagina() {
  if (!huidigePaginaEl) nieuwePagina();
  return huidigePaginaEl;
}

/**
 * Voeg een banner-element (opmaak 2) toe aan de huidige pagina.
 * @param {string} icoon  - emoji of tekst voor het icoongedeelte
 * @param {string} tekst  - titeltekst van de banner
 */
function voegBannerToe(icoon, tekst) {
  const banner = document.createElement('div');
  banner.classList.add('banner');

  const icoonDiv = document.createElement('div');
  icoonDiv.classList.add('banner-icoon');
  icoonDiv.textContent = icoon;

  const tekstDiv = document.createElement('div');
  tekstDiv.classList.add('banner-tekst');
  tekstDiv.textContent = tekst;

  banner.appendChild(icoonDiv);
  banner.appendChild(tekstDiv);
  huidigePagina().appendChild(banner);
  return banner;
}

/**
 * Voeg een paragraaftitel (opmaak 3) toe aan de huidige pagina.
 * @param {string} nummer - bv. '10.1'
 * @param {string} tekst  - bv. 'Eigenschappen van bewerkingen'
 */
function voegParagraafToe(nummer, tekst) {
  const div = document.createElement('div');
  div.classList.add('opmaak-paragraaf');

  const nrSpan = document.createElement('span');
  nrSpan.classList.add('paragraaf-nummer');
  nrSpan.textContent = nummer;

  const txtSpan = document.createElement('span');
  txtSpan.classList.add('paragraaf-titel');
  txtSpan.textContent = tekst;

  div.appendChild(nrSpan);
  div.appendChild(txtSpan);
  huidigePagina().appendChild(div);
  return div;
}

/**
 * Voeg een onderdeeltitel (opmaak 4) toe aan de huidige pagina.
 * @param {string} icoon - emoji of tekst
 * @param {string} tekst - bv. 'Ontdekkingstocht'
 */
function voegOnderdeelToe(icoon, tekst) {
  const div = document.createElement('div');
  div.classList.add('opmaak-onderdeel');

  const icoonSpan = document.createElement('span');
  icoonSpan.classList.add('onderdeel-icoon');
  icoonSpan.textContent = icoon;

  const txtSpan = document.createElement('span');
  txtSpan.classList.add('onderdeel-titel');
  txtSpan.textContent = tekst;

  div.appendChild(icoonSpan);
  div.appendChild(txtSpan);
  huidigePagina().appendChild(div);
  return div;
}

/**
 * Voeg een inhoudszone toe (lege ruimte voor later te vullen)
 * @param {string} id - optionele id voor later gebruik
 */
function voegContentToe(id = '') {
  const div = document.createElement('div');
  div.classList.add('content-ruimte');
  if (id) div.id = id;
  huidigePagina().appendChild(div);
  return div;
}

(function vulInhoudstafel() {
  const container = document.getElementById('inhoudstafel-container');
  if (!container) return;

  const items = [
    { tekst: 'Inhoudstafel',  blz: blzInhoudstafel, stijl: 'bold' },
    { type: 'witregel' },
    { tekst: 'Inchecken',     blz: null },
    { type: 'witregel' },
    { tekst: 'Voorkennis',    blz: null },
    { tekst: 'Succescriteria', blz: null },
    { tekst: 'Overzicht opdrachten', blz: null },
    { type: 'witregel' },
    { tekst: '10.1 Eigenschappen van bewerkingen', blz: null },
    { tekst: '10.2 Lettervormen',                  blz: null },
    { tekst: '10.3 Lettervormen herleiden',         blz: null },
    { tekst: '10.4 Een getal vermenigvuldigen met een lettervorm', blz: null },
    { type: 'witregel' },
    { tekst: 'Schema',        blz: null },
    { type: 'witregel' },
    { tekst: 'Uitchecken',    blz: null },
  ];

  // Koppel paginanummers vanuit inhoudstafelData
  inhoudstafelData.forEach(({ tekst, blz }) => {
    const item = items.find(i => i.tekst === tekst);
    if (item) item.blz = blz;
  });

  items.forEach(item => {
    if (item.type === 'witregel') {
      const br = document.createElement('div');
      br.style.height = '0.5em';
      container.appendChild(br);
      return;
    }

    const rij = document.createElement('div');
    rij.classList.add('inhoudstafel-rij');
    if (item.stijl === 'bold') rij.style.fontWeight = 'bold';

    const tekst = document.createElement('span');
    tekst.classList.add('it-tekst');
    tekst.textContent = item.tekst;
    rij.appendChild(tekst);

    // Stippen (alleen systeem 2)
    const stippen = document.createElement('span');
    stippen.classList.add('it-stippen');
    rij.appendChild(stippen);

    // Paginanummer (alleen systeem 2)
    const blzSpan = document.createElement('span');
    blzSpan.classList.add('it-blz');
    blzSpan.textContent = item.blz !== null ? item.blz : '??';
    rij.appendChild(blzSpan);

    container.appendChild(rij);
  });
})();
