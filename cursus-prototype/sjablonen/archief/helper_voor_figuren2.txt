// TEKST TRANSFORMATIES

function exponent(s) {
  if (!s) return '';
  s = s.replace(/\^\{([^}]*)\}/g, (_, i) => `<sup>${cursief(i)}</sup>`);
  s = s.replace(/\^([a-zA-Z0-9\-])/g, '<sup>$1</sup>');
  return s;
}
function cursief(s) {
  if (!s) return '';
  return s.replace(/\*([^*]+)\*/g, '<i>$1</i>');
}
function opmaak(s) { return s ? cursief(exponent(s)) : ''; }
function escStr(s) { return (s||'').replace(/\\/g,'\\\\').replace(/'/g,"\\'"); }

function splitOpBreuken(str) {
  const parts=[]; const re=/B(\d+)/g; let last=0, m;
  while((m=re.exec(str))!==null){
    if(m.index>last) parts.push({type:'tekst',val:str.slice(last,m.index)});
    parts.push({type:'breuk',idx:parseInt(m[1])});
    last=m.index+m[0].length;
  }
  if(last<str.length) parts.push({type:'tekst',val:str.slice(last)});
  if(parts.length===0) parts.push({type:'tekst',val:str});
  return parts;
}

function bouwJsWaarde(str, stijl) {
  str = str || '';
  if (!/B\d+/.test(str)) return `'${escStr(opmaak(str))}'`;
  const parts = splitOpBreuken(str);
  const isOpl = stijl === 'oplossing';
  const L = [];
  L.push('(function(){');
  L.push("  const s=document.createElement('span');");
  if (isOpl) {
    L.push("  s.style.fontFamily='var(--font-oplossing)';");
    L.push("  s.style.color='var(--blauw-oplossing)';");
  }
  parts.forEach((p, i) => {
    if (p.type === 'tekst') {
      L.push(`  const p${i}=document.createElement('span'); p${i}.innerHTML='${escStr(opmaak(p.val))}'; s.appendChild(p${i});`);
    } else {
      const t  = document.getElementById(`breuk-teller-${p.idx}`)?.value || '?';
      const nm = document.getElementById(`breuk-noemer-${p.idx}`)?.value || '?';
      L.push(`  s.appendChild(maakBreuk('${escStr(t)}','${escStr(nm)}','${stijl}'));`);
    }
  });
  L.push('  return s;})()');
  return L.join('\n    ');
}

function setCode(doel, expr) {
  return `{ const _v=${expr}; if(typeof _v==='string') ${doel}.innerHTML=_v; else ${doel}.appendChild(_v); }`;
}

function activeerRadioGroep(groepEl) {
  groepEl.addEventListener('change', e => {
    if (e.target.type === 'checkbox') {
      if (e.target.value === 'geen' && e.target.checked) {
        groepEl.querySelectorAll('input[type="checkbox"]').forEach(cb => {
          if (cb.value !== 'geen') { cb.checked=false; cb.closest('.radio-optie').classList.remove('geselecteerd'); }
        });
      } else if (e.target.checked) {
        const g = groepEl.querySelector('input[value="geen"]');
        if (g) { g.checked=false; g.closest('.radio-optie').classList.remove('geselecteerd'); }
      }
      e.target.closest('.radio-optie').classList.toggle('geselecteerd', e.target.checked);
    } else {
      groepEl.querySelectorAll('.radio-optie').forEach(o => o.classList.remove('geselecteerd'));
      e.target.closest('.radio-optie').classList.add('geselecteerd');
    }
  });
}
document.querySelectorAll('.radio-groep').forEach(activeerRadioGroep);

function bouwInhoud() {

      </div>

        <span>#</span>
        <input type="number" id="nr-val-opgave-${lbl}" min="1" max="99" value="1">
      </div>`;
    sectieOpg.appendChild(rijOp);
    blok.appendChild(sectieOpg);

  });

  updateStappen(4);
  document.getElementById('inhoud-sectie').scrollIntoView({behavior:'smooth'});
}

function voegSchrijflijnToe(lbl, s, slCont) {
  const rij = document.createElement('div'); rij.classList.add('opl-rij'); rij.id = `rij-opl-${lbl}-${s}`;

  const lbEl = document.createElement('label'); lbEl.className = 'inline';
  lbEl.textContent = `Oplossing ${s+1}:`;

  const inp = document.createElement('input'); inp.type = 'text'; inp.id = `opl-${lbl}-${s}`;
  inp.placeholder = s === 0 ? 'eerste stap' : 'volgende stap'; inp.style.flex = '1';

  const vgWrap = maakVgCheckbox(`vg-${lbl}-${s}`);

  const verkWrap = document.createElement('div'); verkWrap.classList.add('verk-wrap'); verkWrap.id = `verk-opl-${lbl}-${s}`;
  const verkCb   = document.createElement('input'); verkCb.type = 'checkbox'; verkCb.id = `verk-cb-opl-${lbl}-${s}`;
  verkCb.onchange = () => toggleVerkortOpl(lbl, s);
  const verkSp   = document.createElement('span'); verkSp.textContent = 'verkort';
  verkWrap.appendChild(verkCb); verkWrap.appendChild(verkSp);

  const nrWrap = document.createElement('div'); nrWrap.classList.add('nr-wrap'); nrWrap.id = `nr-opl-${lbl}-${s}`;
  const nrSp   = document.createElement('span'); nrSp.textContent = '#';
  const nrInp  = document.createElement('input'); nrInp.type = 'number'; nrInp.id = `nr-val-opl-${lbl}-${s}`;
  nrInp.min = 1; nrInp.max = 99; nrInp.value = 1;
  nrWrap.appendChild(nrSp); nrWrap.appendChild(nrInp);

  rij.appendChild(lbEl); rij.appendChild(inp); rij.appendChild(vgWrap);
  rij.appendChild(verkWrap); rij.appendChild(nrWrap);
  slCont.appendChild(rij);
}

function maakVgCheckbox(cbId) {
  const wrap = document.createElement('label'); wrap.classList.add('vg-wrap');
  const cb   = document.createElement('input'); cb.type = 'checkbox'; cb.id = cbId;
  const sp   = document.createElement('span'); sp.textContent = 'zichtbaar';
  wrap.appendChild(cb); wrap.appendChild(sp);
  cb.addEventListener('change', () => {
    wrap.classList.toggle('actief', cb.checked);
    const rij = cb.closest('.opl-rij');
    const inp = rij?.querySelector('input[type="text"]');
    if (cb.checked) inp?.classList.add('vg-actief');
    else            inp?.classList.remove('vg-actief');
  });
  return wrap;
}

function getVal(id)     { return document.getElementById(id)?.value || ''; }
function isVg(id)       { return document.getElementById(id)?.checked || false; }
function isVerkort(id)  { return document.getElementById(`verk-cb-${id}`)?.checked || false; }
function getNr(id)      { return parseInt(document.getElementById(`nr-val-${id}`)?.value) || 1; }

function verzamelRijen() {
  const rijen   = [];
  const alfabet = 'abcdefghijklmnopqrstuvwxyz';
  const isK0    = sjabloon === 'K0SLT1V';
  const labels  = Array.from({length: aantalPerKolom}, (_, i) => isK0 ? String(i) : alfabet[i]);

  rijen.push({ sleutel:'intro', type:'intro',
    verkort: isVerkort('intro'), nr: getNr('intro') });

  labels.forEach((lbl, idx) => {
    const displayLbl = alfabet[idx];
    rijen.push({ sleutel:`opgave-${lbl}`, type:'opgave', lbl, displayLbl,
      verkort: isVerkort(`opgave-${lbl}`), nr: getNr(`opgave-${lbl}`) });

    const slC = document.getElementById(`sl-container-${lbl}`);
    const n   = slC ? slC.querySelectorAll('.opl-rij').length : 1;
    for (let s = 0; s < n; s++) {
      rijen.push({ sleutel:`opl-${lbl}-${s}`, type:'schrijflijn', lbl, displayLbl, s,
        verkort: isVerkort(`opl-${lbl}-${s}`), nr: getNr(`opl-${lbl}-${s}`),
        vg: isVg(`vg-${lbl}-${s}`) });
    }
  });
  return rijen;
}

function groepeerContainers(rijen) {
  const containers = [];
  const nrNaarLaatste = new Map();
  const dubbele = new Set();
  let huidig = null;

  rijen.forEach((rij, i) => {
    if (!rij.verkort) { huidig = null; return; }
    if (huidig && huidig.nr === rij.nr) {
      huidig.indices.push(i);
    } else {
      if (nrNaarLaatste.has(rij.nr)) dubbele.add(rij.nr);
      huidig = { nr: rij.nr, indices: [i] };
      containers.push(huidig);
      nrNaarLaatste.set(rij.nr, huidig);
    }
  });
  containers.forEach((c, i) => { c.naam = `afbeelding${i + 1}`; });
  return { containers, dubbele: [...dubbele] };
}

function genereer() {
  toonMelding('');
  if (!sjabloon) { toonMelding('Kies eerst een sjabloon.', 'fout'); return; }
  if (!document.getElementById('inhoud-sectie').classList.contains('zichtbaar')) {
    toonMelding('Ga eerst naar de inhoud.', 'fout'); return;
  }

  const hoofdstuk  = getVal('inp-hoofdstuk') || 'H??';
  const paragraaf  = getVal('inp-paragraaf') || 'P??';
  const id         = getVal('inp-id')        || 'oe_???';
  const soortEl    = document.querySelector('input[name="soort"]:checked');
  const soort      = soortEl ? soortEl.value : 'inoefenen';
  const niveauCbs  = document.querySelectorAll('input[name="niveau"]:checked');
  const niveaus    = [...niveauCbs].map(cb => cb.value).filter(v => v !== 'geen');
  const niveausStr = niveaus.length ? `['${niveaus.join("','")}']` : '[]';
  const alfabet    = 'abcdefghijklmnopqrstuvwxyz';
  const isK0       = sjabloon === 'K0SLT1V';
  const labels     = Array.from({length: aantalPerKolom}, (_, i) => isK0 ? String(i) : alfabet[i]);
  const breuk      = heeftBreuken;

  const rijen = verzamelRijen();
  const { containers, dubbele } = groepeerContainers(rijen);

  if (dubbele.length > 0) {
    toonMelding(`Waarschuwing: containernummer(s) ${dubbele.join(', ')} komen voor bij niet-aaneengesloten rijen. Elke aaneengesloten groep krijgt een eigen container.`, 'waarschuwing');
  }

  const rijNaarCont = new Map();
  containers.forEach(cont => cont.indices.forEach(i => rijNaarCont.set(i, cont.naam)));

  // Breedte van de figuur-container = rest naast de verkorte rijen
  const verk = verkorting.trim();
  const figBreedte = verk.endsWith('%')
    ? `calc(${100 - parseFloat(verk)}%)`
    : `calc(100% - ${verk})`;

  let c = '';
  c += `/* ============================================================\n`;
  c += `   ${hoofdstuk} / ${paragraaf} / ${id}.js\n`;
  c += `   soort: ${soort}\n`;

  c += `   ============================================================ */\n\n`;

  c += `(function () {\n\n`;
  c += `  /* Zet RAND_ZICHTBAAR op false om de randen van de figuur-containers te verbergen.\n`;
  c += `     ============================================================ */\n`;
  c += `  const RAND_ZICHTBAAR = true;\n\n`;

  c += `  const inhoud = maakOefening({\n`;
  c += `    id:'${id}', nummer:'?', soort:'${soort}',\n`;
  c += `    niveaus:${niveausStr}, tools:[],\n`;
  c += `    toggleModus:'oefening'\n`;
  c += `  });\n`;
  c += `  inhoud.style.position = 'relative';\n\n`;

  // Hulpfunctie figuur-container
  c += `  function maakFiguurContainer(naam, breedte) {\n`;
  c += `    const div = document.createElement('div');\n`;
  c += `    div.id = naam;\n`;
  c += `    div.style.position      = 'absolute';\n`;
  c += `    div.style.right         = '0';\n`;
  c += `    div.style.width         = breedte;\n`;
  c += `    div.style.display       = 'flex';\n`;
  c += `    div.style.alignItems    = 'center';\n`;
  c += `    div.style.justifyContent = 'center';\n`;
  c += `    div.style.boxSizing     = 'border-box';\n`;
  c += `    if (RAND_ZICHTBAAR) {\n`;
  c += `      div.style.border = '1.5px dashed #aaa';\n`;
  c += `      const lbl = document.createElement('span');\n`;
  c += `      lbl.textContent = naam;\n`;
  c += `      lbl.style.cssText = 'font-size:9pt;color:#999;pointer-events:none;font-family:Arial,sans-serif;';\n`;
  c += `      div.appendChild(lbl);\n`;
  c += `    }\n`;
  c += `    return div;\n`;
  c += `  }\n\n`;

  // Intro
  let rijIdx = 0;
  const introRij  = rijen[rijIdx];
  const introVerk = introRij.verkort;
  const introCont = rijNaarCont.get(rijIdx);
  const introExpr = bouwJsWaarde(getVal('inp-intro'), 'opgave');
  rijIdx++;

  c += `  /* Introductietekst */\n`;
  c += `  {\n`;
  if (introVerk) {
    c += `    const _wrap = document.createElement('div');\n`;
    c += `    _wrap.dataset.figuur = '${introCont}';\n`;
    c += `    const _p = document.createElement('p');\n`;
    c += `    _p.classList.add('intro-tekst');\n`;
    c += `    _p.style.width = '${verk}';\n`;
    c += `    ${setCode('_p', introExpr)}\n`;
    c += `    _wrap.appendChild(_p); inhoud.appendChild(_wrap);\n`;
  } else {
    c += `    const _p = document.createElement('p');\n`;
    c += `    _p.classList.add('intro-tekst');\n`;
    c += `    ${setCode('_p', introExpr)}\n`;
    c += `    inhoud.appendChild(_p);\n`;
  }
  c += `  }\n\n`;

  // Tabel (alleen voor K1; voor K0 wordt per opgave een eigen tabel aangemaakt)
  if (!isK0) {
    c += `  const {tbody} = maakOefTabel(inhoud, 1, false);\n\n`;
  }

  // Opgaven en schrijflijnen
  labels.forEach((lbl, lblIdx) => {
    const displayLbl = alfabet[lblIdx];
    const slC        = document.getElementById(`sl-container-${lbl}`);
    const aantalSl   = slC ? slC.querySelectorAll('.opl-rij').length : 1;

    // Opgave
    const opgRij  = rijen[rijIdx];
    const opgVerk = opgRij.verkort;
    const opgCont = rijNaarCont.get(rijIdx);
    const opgExpr = bouwJsWaarde(getVal(`opgave-${lbl}`), 'opgave');
    rijIdx++;

    if (isK0) {
      // K0SLT1V: opgave als p.intro-tekst, gevolgd door een eigen tabel met schrijflijnen
      c += `  /* Opgave ${lblIdx + 1} */\n`;
      c += `  {\n`;
      if (opgVerk) {
        c += `    const _wrap = document.createElement('div');\n`;
        c += `    _wrap.dataset.figuur = '${opgCont}';\n`;
        c += `    const _p = document.createElement('p');\n`;
        c += `    _p.classList.add('intro-tekst');\n`;
        c += `    _p.style.width = '${verk}';\n`;
        c += `    ${setCode('_p', opgExpr)}\n`;
        c += `    _wrap.appendChild(_p); inhoud.appendChild(_wrap);\n`;
      } else {
        c += `    const _p = document.createElement('p');\n`;
        c += `    _p.classList.add('intro-tekst');\n`;
        c += `    ${setCode('_p', opgExpr)}\n`;
        c += `    inhoud.appendChild(_p);\n`;
      }
      c += `  }\n`;
      // Eigen tabel voor de schrijflijnen van deze opgave
      c += `  const {tbody: tbody_${lbl}} = maakOefTabel(inhoud, 1, false);\n\n`;
    } else {
      // K1SLT1V: opgave via maakOpTdaz
      c += `  /* Opgave ${displayLbl} */\n`;
      c += `  {\n`;
      c += `    const _trLbl = document.createElement('tr');\n`;
      if (opgVerk) c += `    _trLbl.dataset.figuur = '${opgCont}';\n`;
      c += `    const _tdLbl = maakOpTdaz('${displayLbl}', ${opgExpr}, ${breuk});\n`;
      if (opgVerk) {
        // Beperk de rij-breedte zodat er ruimte is voor de figuur
        c += `    _tdLbl.querySelector('.rij').style.maxWidth = '${verk}';\n`;
      }
      c += `    /* zet bovenaan */\n`;
      c += `    _tdLbl.querySelector('.rij').style.alignItems = 'flex-start';\n`; 
      c += `    /* geef extra ruimte boven */\n`;
      c += `    _tdLbl.querySelector('.lbl').style.paddingTop = '0cm';\n`; 
      c += `    _tdLbl.classList.add('kolom-links');\n`;
      c += `    _trLbl.appendChild(_tdLbl); tbody.appendChild(_trLbl);\n`;
      c += `  }\n\n`;
    }

    // Schrijflijnen
    for (let s = 0; s < aantalSl; s++) {
      const slRij   = rijen[rijIdx];
      const slVerk  = slRij.verkort;
      const slCont2 = rijNaarCont.get(rijIdx);
      const vgb     = slRij.vg;
      const oplExpr = bouwJsWaarde(getVal(`opl-${lbl}-${s}`), 'oplossing');
      const vTd     = `_td_${lbl}_${s}`;
      const vTr     = `_tr_${lbl}_${s}`;
      rijIdx++;

      c += `  /* Schrijflijn ${s + 1} van ${isK0 ? lblIdx + 1 : displayLbl} */\n`;
      c += `  {\n`;
      c += `    const ${vTr} = document.createElement('tr');\n`;
      if (slVerk) c += `    ${vTr}.dataset.figuur = '${slCont2}';\n`;

      if (isK0) {
        // maakSlTdZonderInsprong — geen tab, geen insprong
        c += `    const ${vTd} = maakSlTdZonderInsprong(${oplExpr}, ${breuk});\n`;
      } else {
        // maakSlTd — met tab en insprong
        c += `    const ${vTd} = maakSlTd(${oplExpr}, ${breuk});\n`;
      }
      if (vgb) c += `    ${vTd}.querySelector('.opl').classList.add('zichtbaar','opl-vast');\n`;

      // Verkorting op de schrijflijn: begrens de sl-span
      if (slVerk) {
        c += `    ${vTd}.querySelector('.sl').style.maxWidth = 'calc(${verk} - var(--insprong-stap))';\n`;
        c += `    ${vTd}.querySelector('.sl').style.flex = 'none';\n`;
      }

      c += `    ${vTd}.classList.add('kolom-links');\n`;
      // K0: gebruik de tabel van deze opgave; K1: gebruik de gedeelde tabel
      c += `    ${vTr}.appendChild(${vTd}); ${isK0 ? `tbody_${lbl}` : 'tbody'}.appendChild(${vTr});\n`;
      c += `  }\n\n`;
    }
  });

  // Figuur-containers
  if (containers.length > 0) {
    c += `  /* ============================================================\n`;
    c += `     FIGUUR-CONTAINERS\n`;
    c += `     Positie en hoogte worden na render berekend via requestAnimationFrame.\n`;
    c += `     De containers zoeken hun rijen op via het data-figuur attribuut.\n`;
    c += `     ============================================================ */\n\n`;

    containers.forEach(cont => {
      c += `  /* ${cont.naam} */\n`;
      c += `  {\n`;
      c += `    const _fig = maakFiguurContainer('${cont.naam}', '${figBreedte}');\n`;
      c += `    requestAnimationFrame(() => {\n`;
      c += `      const _els = [...inhoud.querySelectorAll('[data-figuur="${cont.naam}"]')];\n`;
      c += `      if (_els.length === 0) return;\n`;
      c += `      const _base   = inhoud.getBoundingClientRect();\n`;
      c += `      const _top    = _els[0].getBoundingClientRect().top    - _base.top    + inhoud.scrollTop;\n`;
      c += `      const _bottom = _els[_els.length-1].getBoundingClientRect().bottom - _base.top + inhoud.scrollTop;\n`;
      c += `      _fig.style.top    = _top + 'px';\n`;
      c += `      _fig.style.height = (_bottom - _top) + 'px';\n`;
      c += `      inhoud.appendChild(_fig);\n`;
      c += `    });\n`;
      c += `  }\n\n`;
    });
  }

  c += `})();\n`;

  document.getElementById('output-code').textContent = c;
  document.getElementById('output-kaart').classList.add('zichtbaar');
  updateStappen(5);
  document.getElementById('output-kaart').scrollIntoView({behavior:'smooth'});
  if (dubbele.length === 0) toonMelding('Code gegenereerd!', 'ok');
}

  }
