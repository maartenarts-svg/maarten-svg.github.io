function verplaatsTerm(n, richting) {
  const blok = document.getElementById(`term-${n}`);
  const cont = document.getElementById('termen-container');
  const blokken = [...cont.querySelectorAll('.term-blok')];
  const idx = blokken.indexOf(blok);
  const nieuwIdx = idx + richting;
  if (nieuwIdx < 0 || nieuwIdx >= blokken.length) return;
  if (richting === -1) {
    cont.insertBefore(blok, blokken[nieuwIdx]);
  } else {
    cont.insertBefore(blokken[nieuwIdx], blok);
  }
  hernoemTermen();
}

function verwijderTerm(n) {
  if (!confirm('Term verwijderen?')) return;
  const el = document.getElementById(`term-${n}`);
  if (el) el.remove();
  hernoemTermen();
}

function hernoemTermen() {
  document.querySelectorAll('.term-blok').forEach((blok, i) => {
    blok.querySelector('.term-nr').textContent = `Term ${i + 1}`;
  });
}

function exponent(s) {
  s = s.replace(/\^\{([^}]*)\}/g, (_, i) => `<sup>${cursief(i)}</sup>`);
  s = s.replace(/\^([a-zA-Z0-9\-])/g, '<sup>$1</sup>');
  return s;
}
function cursief(s) {
  return s.replace(/\*([^*]+)\*/g, '<i>$1</i>');
}
function opmaak(s) { return s ? cursief(exponent(s)) : ''; }
function escStr(s) { return (s||'').replace(/\\/g,'\\\\').replace(/'/g,"\\'"); }

function getBreukWaarde(idx, stijl) {
  const t  = document.getElementById(`breuk-teller-${idx}`)?.value || '?';
  const nm = document.getElementById(`breuk-noemer-${idx}`)?.value || '?';
  return { t, nm, stijl };
}

// ============================================================
// INHOUD PARSEN
// ============================================================
/*
  Ondersteunde patronen in inhoud-string:
  - $tekst$ вҶ’ zetOmNaarOpl
  - $B1$    вҶ’ breuk in opl
  - ['tekst','#kleur',true/false] вҶ’ markering
  - ВЈbreedte,'tekst'ВЈ of ВЈbreedteВЈ вҶ’ maakSpanBreedte
  - B1      вҶ’ breuk (gewone stijl)
  - rest    вҶ’ htmlSpan met opmaak
*/
function parseInhoud(str, varNaam, varTeller) {
  const lijnen = [];
  let rest = str.trim();

  // Controleer of het een markering is: ['...','...', bool]
  const markMatch = rest.match(/^\['(.*)','(.*)'\s*,\s*(true|false)\]$/);
  const markMatchDefault = rest.match(/^\['(.*)'\]$/);
  const markMatchKleur = rest.match(/^\['(.*)','(.*)'\]$/);
  const markBreukMatch = rest.match(/^\[B(\d+),'(.*)'\s*,\s*(true|false)\]$/);
  const markBreukDefault = rest.match(/^\[B(\d+)\]$/);
  const markBreukKleur = rest.match(/^\[B(\d+),'(.*)'\]$/);

  if (markBreukMatch || markBreukDefault || markBreukKleur) {
    let idx, kleur, toggle;
    if (markBreukMatch) {
      idx    = parseInt(markBreukMatch[1]);
      kleur  = markBreukMatch[2];
      toggle = markBreukMatch[3] === 'true';
    } else if (markBreukKleur) {
      idx    = parseInt(markBreukKleur[1]);
      kleur  = markBreukKleur[2];
      toggle = true;
    } else {
      idx    = parseInt(markBreukDefault[1]);
      kleur  = '#bbdefb';
      toggle = true;
    }
    const { t, nm } = getBreukWaarde(idx, 'opgave');
    const vn = `${varNaam}_m${varTeller}`;
    lijnen.push(`const ${vn} = maakMarkering(maakBreuk('${escStr(t)}','${escStr(nm)}','opgave'), '${kleur}', ${toggle});`);
    return { lijnen, ref: vn };
  }

  if (markMatch || markMatchDefault || markMatchKleur) {
    let tekst, kleur, toggle;
    if (markMatch) {
      tekst  = markMatch[1];
      kleur  = markMatch[2];
      toggle = markMatch[3] === 'true';
    } else if (markMatchKleur) {
      tekst  = markMatchKleur[1];
      kleur  = markMatchKleur[2];
      toggle = true;
    } else {
      tekst  = markMatchDefault[1];
      kleur  = '#bbdefb';
      toggle = true;
    }
    const vn = `${varNaam}_m${varTeller}`;

    // NIEUW: controleer of tekst $...$ tokens bevat
    const tokens = tokeniseer(tekst);
    if (tokens.length === 1 && !tokens[0].opl) {
      // Eenvoudige tekst, zoals vroeger
      lijnen.push(`const ${vn} = maakMarkering(htmlSpan('${escStr(opmaak(tekst))}'), '${kleur}', ${toggle});`);
    } else {
      // Gemengde inhoud: bouw eerst een span op
      const sn = `${vn}_s`;
      lijnen.push(`const ${sn} = document.createElement('span');`);
      tokens.forEach(tok => {
        if (tok.opl) {
          lijnen.push(`${sn}.appendChild(zetOmNaarOpl('${escStr(opmaak(tok.val))}'));`);
        } else {
          lijnen.push(`${sn}.appendChild(htmlSpan('${escStr(opmaak(tok.val))}'));`);
        }
      });
      lijnen.push(`const ${vn} = maakMarkering(${sn}, '${kleur}', ${toggle});`);
    }
    return { lijnen, ref: vn };
  }

  // Controleer breedte-span: ВЈbreedte,'tekst'ВЈ of ВЈbreedteВЈ
  const breedteMatch = rest.match(/^ВЈ([^,ВЈ]+),'([^']*)'ВЈ$/) || rest.match(/^ВЈ([^,ВЈ]+)ВЈ$/);
  if (breedteMatch) {
    const breedte = breedteMatch[1].trim();
    const inhoudTekst = breedteMatch[2] ? escStr(opmaak(breedteMatch[2])) : '';
    const vn = `${varNaam}_b${varTeller}`;
    lijnen.push(`const ${vn} = maakSpanBreedte('${breedte}', '${inhoudTekst}');`);
    return { lijnen, ref: vn };
  }

  // Splits op $...$ en B\d+ patronen
  const tokens = tokeniseer(rest);

  if (tokens.length === 1 && tokens[0].type === 'tekst' && !tokens[0].opl) {
    // Eenvoudige string
    return { lijnen: [], ref: `htmlSpan('${escStr(opmaak(tokens[0].val))}')` };
  }

  const vn = `${varNaam}_s${varTeller}`;
  lijnen.push(`const ${vn} = document.createElement('span');`);

  tokens.forEach((tok, i) => {
    if (tok.type === 'tekst') {
      if (tok.opl) {
        lijnen.push(`${vn}.appendChild(zetOmNaarOpl('${escStr(opmaak(tok.val))}'));`);
      } else {
        lijnen.push(`${vn}.appendChild(htmlSpan('${escStr(opmaak(tok.val))}'));`);
      }
    } else if (tok.type === 'breuk') {
      const stijl = tok.opl ? 'oplossing' : 'opgave';
      const { t, nm } = getBreukWaarde(tok.idx, stijl);
      if (tok.opl) {
        lijnen.push(`${vn}.appendChild(zetOmNaarOpl(maakBreuk('${escStr(t)}','${escStr(nm)}','${stijl}')));`);
      } else {
        lijnen.push(`${vn}.appendChild(maakBreuk('${escStr(t)}','${escStr(nm)}','${stijl}'));`);
      }
    }
  });

  return { lijnen, ref: vn };
}

function tokeniseer(str) {
  const tokens = [];
  // Patronen: $B\d+$, $...$, B\d+
  const re = /\$B(\d+)\$|\$([^$]+)\$|B(\d+)/g;
  let last = 0, m;
  while ((m = re.exec(str)) !== null) {
    if (m.index > last) {
      tokens.push({ type: 'tekst', val: str.slice(last, m.index), opl: false });
    }
    if (m[1]) {
      // $B1$ вҶ’ breuk in opl
      tokens.push({ type: 'breuk', idx: parseInt(m[1]), opl: true });
    } else if (m[2]) {
      // $tekst$ вҶ’ tekst in opl
      tokens.push({ type: 'tekst', val: m[2], opl: true });
    } else if (m[3]) {
      // B1 вҶ’ breuk gewoon
      tokens.push({ type: 'breuk', idx: parseInt(m[3]), opl: false });
    }
    last = m.index + m[0].length;
  }
  if (last < str.length) {
    tokens.push({ type: 'tekst', val: str.slice(last), opl: false });
  }
  if (tokens.length === 0) tokens.push({ type: 'tekst', val: str, opl: false });
  return tokens;
}

function genereer() {
  toonMelding('');
  const varNaam = document.getElementById('inp-var').value.trim() || 'h1';
  const termBlokken = document.querySelectorAll('.term-blok');
  const aantalSpaties = parseInt(document.getElementById('inp-insprong').value) || 0;
  const ins = ' '.repeat(aantalSpaties);

  if (termBlokken.length === 0) {
    toonMelding('Voeg minstens Г©Г©n term toe.', 'fout');
    return;
  }

  const lijnen = [];
  lijnen.push(`const ${varNaam} = document.createElement('span');`);

  termBlokken.forEach((blok, i) => {
    const id = blok.id.replace('term-', '');
    const inhoudStr = document.getElementById(`inhoud-${id}`).value.trim();
    const stijl     = document.getElementById(`stijl-${id}`).value;
    const diepte    = document.getElementById(`diepte-${id}`)?.value.trim() || '0.1cm';
    const hoogte    = document.getElementById(`hoogte-${id}`)?.value.trim() || '';
    const toggle    = document.getElementById(`toggle-${id}`)?.checked ?? true;

    if (!inhoudStr === '') return;

    if (inhoudStr.trim() === '') {
      lijnen.push(`${varNaam}.appendChild(htmlSpan(' '));`);
      return;
    }

    const { lijnen: subLijnen, ref } = parseInhoud(inhoudStr, varNaam, i + 1);
    lijnen.push(...subLijnen);

    if (stijl === 'geen') {
      lijnen.push(`${varNaam}.appendChild(${ref});`);
    } else {
      const hoogteParam = hoogte ? `, '${hoogte}'` : '';
      lijnen.push(`${varNaam}.appendChild(maakOnderstreping(${ref}, '${stijl}', '${diepte}', ${toggle}${hoogteParam}));`);
    }
  });

  const code = lijnen.map(l => ins + l).join('\n');
  document.getElementById('output-code').textContent = code;
  document.getElementById('output-kaart').classList.add('zichtbaar');
  document.getElementById('output-kaart').scrollIntoView({ behavior: 'smooth' });
  toonMelding('Code gegenereerd!', 'ok');
}
